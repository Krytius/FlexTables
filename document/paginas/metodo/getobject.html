getObject()

Método que obtém o objeto JSON no momento atual.

Retorno

  • JSON

Exemplo

var grid = new Grid('elementoID');
grid.setColunas(['Produto', 'Preço', 'Quantidade']);
grid.setLarguraColunas(['100', '*', '100']);
grid.setLarguraColunas(['str', 'str', 'int']);
grid.setColunasEventos(['noEvent', 'noEvent', 'noEvent']);
grid.init({
    "rows": [
        {
            "id": "864", "data": ["Produto 1", "10.00", "2"]
        },
        {
            "id": "865", "data": ["Produto 2", "0.00", "0"]
        }
    ]
});

console.log(grid.getObject());
/*
{
    "rows": [
        {
            "id": "864", "data": ["Produto 1", "10.00", "2"]
        },
        {
            "id": "865", "data": ["Produto 2", "0.00", "0"]
        }
    ]
}
*/
